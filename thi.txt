private void addValuesToCBO()
        {
            DataTable dta = new DataTable();
            dta = ketnoi.Lay_Dulieu("select * from DIEUCHUYEN");
            cboMADC.DataSource = dta;
            cboMADC.DisplayMember = "MADC";
            cboMADC.ValueMember = "MADC";

            cboMATS.DataSource = dta;
            cboMATS.DisplayMember = "MATS";
            cboMATS.ValueMember = "MATS";

            cboMADONIV_DEN.DataSource = dta;
            cboMADONIV_DEN.DisplayMember = "MADONIV_DEN";
            cboMADONIV_DEN.ValueMember = "MADONIV_DEN";

            cboMADONVI_DI.DataSource = dta;
            cboMADONVI_DI.DisplayMember = "MADONVI_DI";
            cboMADONVI_DI.ValueMember = "MADONVI_DI";

            cboTINHTRANG.DataSource = dta;
            cboTINHTRANG.DisplayMember = "MATS";
            cboTINHTRANG.ValueMember = "MATS";
        }

public void DATABINDING()
        {
            cboMADC.DataBindings.Clear();
            cboMADC.DataBindings.Add("Text", GridView_DIEUCHUYEN.DataSource, "MADC");

            cboMATS.DataBindings.Clear();
            cboMATS.DataBindings.Add("Text", GridView_DIEUCHUYEN.DataSource, "MATS");

            cboMADONVI_DEN.DataBindings.Clear();
            cboMADONVI_DEN.DataBindings.Add("Text", GridView_DIEUCHUYEN.DataSource, "MADONVI_DEN");

            cboMADONVI_DI.DataBindings.Clear();
            cboMADONVI_DI.DataBindings.Add("Text", GridView_DIEUCHUYEN.DataSource, "MADONVI_DI");

            cboTINHTRANG.DataBindings.Clear();
            cboTINHTRANG.DataBindings.Add("Text", GridView_DIEUCHUYEN.DataSource, "TINHTRANG");

        }
